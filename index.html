Structure by block : 
LOOP := for NUMBER times do
	  STATEMENT+
	endfor
PROGRAM :=
(FUNCTION|PROCEDURE)*
algo IDENTIFIER
DECLARATION*
begin
  STATEMENT+
end
Example of loop block : 
for 10 times
  a = a+1;
endfor
Example of program structure : 
algo HelloWorld
begin
 afficher("Hello World !")
end

Program segmentation :

FUNCTION :=
function TYPE IDENTIFIER(  (PARAM(,PARAM)*)? )
DECLARATION*
begin
  STATEMENT*
  return ACTION
end
Example : 
function int fonction1(int annee)
int ret;
begin
  ret = annee+1;
  afficher("Bonne année !");
  return ret;
end
Statements : 
STATEMENT := AFFECTATION | IFBLOC | LOOP | PROCCALL
LOOP := while (LOGICEXPRESSION)
do
  STATEMENT+
endwhile
Operation : 
OPERATION :=
EXPARITHMETIC | EXPLOGIC | EXPSTRING | FUNCALL | VARIABLE
EXPSTRING := (VARIABLE |"(CHAR|DIGIT)*") (. EXPSTRING)?
